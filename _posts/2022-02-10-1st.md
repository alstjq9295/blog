# ElasticSearch Query example

### Create index
```yaml
PUT /test_index_{YYMM}
{
  "settings": {
    "number_of_shards": "3",
    "number_of_replicas": "1",
    "analysis": {
      "tokenizer": {
        "nori_mixed": {
          "type": "nori_tokenizer",
          "decompound_mode": "mixed"
        }
      },
        "analyzer": {
          "nori_analyzer": {
            "type": "custom",
            "tokenizer": "nori_mixed",
      "filter": ["lowercase"]
    },
    "whitespace": {
      "tokenizer": "whitespace",
      "filter": ["lowercase"],
      "char_filter": ["my_char_filter"]
    }
      },
      "char_filter": {
        "my_char_filter": {
          "type": "mapping",
          "mappings": [
            ", => \\u0020",
            "| => \\u0020"
          ]
        }
      }
    }
  },
  "mappings": {
    "properties": {
      "mbti": {
        "type": "keyword"
      },
      "channel_name": {
        "type": "keyword"
      },
      "url": {
        "type": "keyword"
      },
      "data_date": {
        "type": "date"
      },
      "title": {
        "analyzer": "nori_analyzer",
        "type": "text"
      },
      "content": {
        "analyzer": "nori_analyzer",
        "type": "text"
      },
      "category": {
        "type": "keyword"
      },
      "male_score": {
        "type": "float"
      },
      "female_score": {
        "type": "float"
      },
      "crawled_at": {
        "type": "date"
      }
    }
  }
}
```

### Create Template
```yaml
PUT _template/test_template
{
  "index_patterns": [
    "test_index_*"
  ],
  "settings": {
    "index": {
      "analysis": {
        "analyzer" : {
          "whitespace": {
            "filter": [
              "lowercase"
            ],
            "char_filter": [
              "my_char_filter"
            ],
            "tokenizer": "whitespace"
          },
          "nori_analyzer": {
            "filter": [
              "lowercase"
            ],
            "type": "custom",
            "tokenizer": "nori_mixed"
          }
        },
        "char_filter": {
          "my_char_filter": {
            "type": "mapping",
            "mappings": [
              """, => \u0020""",
              """| => \u0020"""
            ]
          }
        },
        "tokenizer": {
          "nori_mixed": {
            "type": "nori_tokenizer",
            "decompound_mode": "mixed"
          }
        }
      },
      "number_of_shards": "3",
      "number_of_replicas": "1"
    }
  },
  "mappings": {
    "properties": {
      "mbti": {
        "type": "keyword"
      },
      "channel_name": {
        "type": "keyword"
      },
      "url": {
        "type": "keyword"
      },
      "data_date": {
        "type": "date"
      },
      "title": {
        "type": "text",
        "fields": {
          "nori": {
            "type": "text",

          },
          "whitespace": {
            "type": "text",
            "analyzer": "whitespace"
          }
        }
      },
      "content": {
        "type": "text",
        "fields": {
          "nori": {
            "type": "text",

            "type": "text",
```

### Count group by
```yaml
GET test_mbti/_search
{
  "size": 0,
  "query": {
    "bool": {
      "filter": [
        {
          "range": {
            "data_date": {
              "gte": "2022-01-01",
              "lte": "2022-01-31"
            }
          }
        }
      ]
    }
  },
  "aggs": {
    "by_mbti": {
      "terms": {
        "field": "mbti"
      }
    }
  }
}
```

### Count group by with text size
```yaml
GET test_mbti/_search
{
  "size": 0,
  "query": {
    "bool": {
      "filter": [
        {
          "range": {
            "data_date": {
              "gte": "2022-01-01",
              "lte": "2022-01-31"
            }
          }
        },
        {
          "script": {
            "script": {
              "source": "doc['content'].size() > 150"
            }
          }
        }
      ]
    }
  },
  "aggs": {
    "by_mbti": {
      "terms": {
        "field": "mbti"
      }
    }
  }
}
```

### Mapping
```yaml
PUT [INDEX]/_mapping
{
  "properties": {
    "[FIELD]": {
      ["analyzer": "[ANALYZER]",] # Analyzer 있는 필드면
      "type": "[FIELD TYPE]",
      "fielddata": true
    }
  }
}
```
