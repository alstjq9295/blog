## MBTI 별 데이터 aggregation query
```yaml
GET mbti_naver_blog/_search
{
  "size": 5,
  "sort": [
    {
      "reg_date": {
        "order": "desc"
      }
    }
  ],
  "query": {
    "bool": {
      "filter": {
        "range": {
          "reg_date": {
            "gte": "YYYY-mm-dd",
            "lte": "YYYY-mm-dd"
          }
        }
      },
      "should": [
        {
          "match_phrase": {
            "title.nori_mixed": "의사"
          }
        },{
          "match_phrase": {
            "title.nori_noun": "의사"
          }
        },{
          "match_phrase": {
            "contents.nori_mixed": "의사"
          }
        },{
          "match_phrase": {
            "contents.nori_noun": "의사"
          }
        }
      ],
      "minimum_should_match": 1
    }
  },
  "aggs": {
    "by_mbti": {
      "terms": {
        "field": "keyword"
      }
    }
  }
}
```
